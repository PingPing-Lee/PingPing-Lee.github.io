在三维图形学中，有一个名词叫做 纹理映射，就是将一张图形(就像一张贴纸)映射(贴)到一个几何图形的表面，此时这张图片又可以成为 纹理图像 或 纹理

纹理映射 的作用，就是根据纹理图像，为之前光栅化后的每个片元涂上合适的颜色。

组成纹理图像的像素又被称为 纹素，每个纹素的颜色都使用 RGB 或 RGBA 格式编码

在 WebGL 中，要进行纹理映射，需要遵循以下四步：

  • 准备好映射到几何图形上的纹理图像
  • 为几何图形配置纹理映射方式
  • 加载纹理图像，对其进行一些配置，在WebGL 中使用它
  • 在片元着色器中将相应的纹素从纹理中抽取出来，并将纹素的颜色赋给片元

这其中第一步很好理解，第二步指定映射方式，是利用图形的顶点坐标来确定屏幕上哪部分被纹理图像覆盖，使用 纹理坐标 来确定纹理图像的哪部分将覆盖到几何图形上。

# 纹理坐标

纹理坐标 是一套新的坐标系统，是纹理图像上的坐标，通过纹理坐标可以在纹理图像上获取纹素颜色。

纹理坐标系统 虽然也是二维的，但是和 X Y 坐标又有所不同，所以为了区分开，WebGL 中使用 s 和t 命名纹理坐标(st坐标系统)，除了命名以外，竖轴的方向也是相反的，下图就是 WebGL 的纹理坐标系统：

# 配置和加载纹理

理清楚纹理坐标和顶点坐标映射关系之后，我们来创建配置纹理，示例代码

    // 片元着色器中定义
    uniform sampler2D uSampler;
    // 获取变量，用来接收纹理图像
    const uSampler = gl.getUniformLocation(program, 'uSampler');
    // 创建纹理对象
    const texture = gl.createTexture();
    //  对纹理图像 Y轴翻转
    gl.pixelStorei(gl.UNPACK_FLIP_Y_WEBGL, 1);

    // 开启一个纹理单元
    gl.activeTexture(gl.TEXTURE0);

    // 绑定纹理对象
    gl.bindTexture(gl.TEXTURE_2D, texture);

    // 配置纹理参数(处理放大缩小的逻辑)
    gl.texParameteri(gl.TEXTURE_2D, gl.TEXTURE_MAG_FILTER, gl.LINEAR)
    gl.texParameteri(gl.TEXTURE_2D, gl.TEXTURE_MIN_FILTER, gl.LINEAR)

    // 配置纹理参数(横向 纵向 平铺的方式)
    gl.texParameteri(gl.TEXTURE_2D, gl.TEXTURE_WRAP_S, gl.CLAMP_TO_EDGE)
    gl.texParameteri(gl.TEXTURE_2D, gl.TEXTURE_WRAP_T, gl.CLAMP_TO_EDGE)

    // 配置纹理图像
    gl.texImage2D(gl.TEXTURE_2D, 0, gl.RGB, gl.RGB, gl.UNSIGNED_BYTE, img);

    // 将纹理传递给着色器中的取样器变量
    gl.uniform1i(uSampler, 0);

  • 通过 uniform sampler2D uSampler 来声明，其中 sampler2D 是一种专用于纹理的数据类型

  • gl.createTexture() 来创建，同样可以通过 gl.deleteTexture(texture) 来删除纹理对象

  • 图像通过 gl.pixelStorei(gl.UNPACK_FLIP_Y_WEBGL, 1) 进行Y 轴反转，因为 st 坐标系的竖轴和 XY坐标系是相反的

    方法规范如下：

  • WebGL 是通过纹理单元来管理纹理对象，每个纹理单元管理⼀张纹理图像，在使用之前需要通过 gl.activeTexture(gl.TEXTURE0) 激活，方法参数是指定准备激活的纹理单元

  • 通过 gl.bindTexture(gl.TEXTURE_2D, texture) 告诉 WebGL 系统纹理对象使用的是哪种类型的纹理，其中参数 gl.TEXTURE_2D 代表的是二维纹理，还有一个立方体纹理选项 gl.TEXTURE_CUBE_MAP

  • 通过 gl.texParameteri(target, pname, param) 配置纹理对象的参数

    • 这其中的 pname 纹理参数有四个选项
    纹理参数 描述
    gl.TEXTURE_MAG_FILTER 纹理放⼤
    gl.TEXTURE_MIN_FILTER 纹理缩⼩
    gl.TEXTURE_WRAP_S 横向（⽔平填充）
    gl.TEXTURE_WRAP_T 纵向（垂直填充）

# 绘图

最后一步就是把图片绘制出来，沿用 canvas 的绘图思路，整体代码如下：

<script>
  const ctx = document.getElementById('canvas')
  const gl = ctx.getContext('webgl')
  const VERTEX_SHADER_SOURCE = `
    attribute vec4 aPosition;
    attribute vec4 aTex;
    varying vec2 vTex;
    void main() {
      gl_Position = aPosition;
      vTex = vec2(aTex.x, aTex.y);
    }
  `;
  const FRAGMENT_SHADER_SOURCE = `
    precision lowp float;
    uniform sampler2D uSampler;
    varying vec2 vTex;
    void main() {
        gl_FragColor = texture2D(uSampler, vTex);
    }
  `;
  const program = initShader(gl, VERTEX_SHADER_SOURCE, FRAGMENT_SHADER_SOURCE)
  const aPosition = gl.getAttribLocation(program, 'aPosition');
  const aTex = gl.getAttribLocation(program, 'aTex');
  const uSampler = gl.getUniformLocation(program, 'uSampler');
  const points = new Float32Array([
    -0.5,  0.5, 0.0, 1.0,
    -0.5, -0.5, 0.0, 0.0,
     0.5,  0.5, 1.0, 1.0,
     0.5, -0.5, 1.0, 0.0,
  ])
  const buffer = gl.createBuffer();
  const BYTES = points.BYTES_PER_ELEMENT;
  gl.bindBuffer(gl.ARRAY_BUFFER, buffer)
  gl.bufferData(gl.ARRAY_BUFFER, points, gl.STATIC_DRAW)
  gl.vertexAttribPointer(aPosition, 2, gl.FLOAT, false, BYTES * 4, 0)
  gl.enableVertexAttribArray(aPosition)
  gl.vertexAttribPointer(aTex, 2, gl.FLOAT, false, BYTES * 4, BYTES * 2);
  gl.enableVertexAttribArray(aTex)
  const img = new Image();
    img.onload = function() {
        const texture = gl.createTexture();
        gl.pixelStorei(gl.UNPACK_FLIP_Y_WEBGL, 1)
        gl.activeTexture(gl.TEXTURE0);
        gl.bindTexture(gl.TEXTURE_2D, texture);
        gl.texParameteri(gl.TEXTURE_2D, gl.TEXTURE_MAG_FILTER, gl.LINEAR)
        gl.texParameteri(gl.TEXTURE_2D, gl.TEXTURE_MIN_FILTER, gl.LINEAR)
        gl.texParameteri(gl.TEXTURE_2D, gl.TEXTURE_WRAP_S, gl.CLAMP_TO_EDGE)
        gl.texParameteri(gl.TEXTURE_2D, gl.TEXTURE_WRAP_T, gl.CLAMP_TO_EDGE)
        gl.texImage2D(gl.TEXTURE_2D, 0, gl.RGB, gl.RGB, gl.UNSIGNED_BYTE, img);
        gl.uniform1i(uSampler, 0);
        // 绘图
        gl.drawArrays(gl.TRIANGLE_STRIP, 0, 4);
    }
    img.src = 'http://127.0.0.1:8081/assets/avatar.png';
</script>
